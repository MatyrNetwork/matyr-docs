# matyr_key

## set keyboard

```js copy
local keyboard = exports["matyr_key"]:key({
    header = "Add Items",
    rows = {
        {
            id = 0,
            txt = "Spawn Name"
        },
        {
            id = 1,
            txt = "Amount"
        }
    }
})

if keyboard ~= nil then
    if keyboard[1].input == nil or keyboard[2].input == nil then return end
    TriggerEvent('additem', keyboard[1].input, keyboard[2].input)
end
```
